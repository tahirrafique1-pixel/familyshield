    // init map
    function initMap() {
        if(mapInstance) return;
        mapInstance = L.map('map', { zoomControl: false }).setView([30.3753, 69.3451], 5);
        L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
            attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OSM</a>'
        }).addTo(mapInstance);
        L.control.zoom({ position: 'bottomright' }).addTo(mapInstance);
    }

    // Helper messages
    function setMessage(msg, isError = false) {
        const msgDiv = document.getElementById('msg');
        if(msgDiv) {
            msgDiv.innerText = msg;
            msgDiv.style.color = isError ? "#ff9f6e" : "#b9fbc0";
            setTimeout(() => {
                if(document.getElementById('msg')?.innerText === msg) 
                    msgDiv.innerText = "";
            }, 4000);
        }
    }

    // Tab switching
    function setTab(t) {
        activeTab = t;
        document.getElementById('login-form').classList.toggle('hidden', t !== 'login');
        document.getElementById('reg-form').classList.toggle('hidden', t !== 'reg');
        const btns = document.querySelectorAll('.tab-btn');
        if(btns[0]) btns[0].classList.toggle('tab-active', t === 'login');
        if(btns[1]) btns[1].classList.toggle('tab-active', t === 'reg');
        setMessage("");
        if(t === 'reg') {
            // reset image preview & data
            selectedPhotoBase64 = "";
            const previewDiv = document.getElementById('imagePreviewCircle');
            if(previewDiv) {
                previewDiv.innerHTML = `<span style="font-size: 2rem;">📷</span>`;
            }
            document.getElementById('r-photo-input').value = "";
        }
    }

    // Handle image upload -> base64 & preview
    function handleImageUpload(inputElement) {
        const file = inputElement.files[0];
        if(!file) {
            selectedPhotoBase64 = "";
            const previewDiv = document.getElementById('imagePreviewCircle');
            if(previewDiv) previewDiv.innerHTML = `<span style="font-size: 2rem;">📷</span>`;
            return;
        }
        if(!file.type.match('image.*')) {
            setMessage("Please select an image file (JPG/PNG)", true);
            inputElement.value = "";
            return;
        }
        const reader = new FileReader();
        reader.onload = (e) => {
            selectedPhotoBase64 = e.target.result;
            // update preview circle
            const previewContainer = document.getElementById('imagePreviewCircle');
            if(previewContainer) {
                previewContainer.innerHTML = `<img src="${selectedPhotoBase64}" alt="preview" style="width:100%;height:100%;object-fit:cover;">`;
            }
        };
        reader.onerror = () => {
            setMessage("Failed to read image", true);
            selectedPhotoBase64 = "";
        };
        reader.readAsDataURL(file);
    }

    // ----- GEOLOCATION (browser) + send location to server periodically (if logged in)
    let lastSentLocation = { lat: 0, lng: 0 };
    function startLocationTracking() {
        if(!navigator.geolocation) return;
        if(locationWatchId !== null) navigator.geolocation.clearWatch(locationWatchId);
        locationWatchId = navigator.geolocation.watchPosition(
            (position) => {
                const lat = position.coords.latitude;
                const lng = position.coords.longitude;
                lastSentLocation = { lat, lng };
                if(currentUser && !currentUser.isAdmin) {
                    sendLocationToServer(lat, lng);
                }
                // update map if we have markers?
                if(mapInstance && currentUser && !currentUser.isAdmin) {
                    // just update own marker later (handled by sync)
                }
            },
            (err) => { console.warn("Geo error:", err.message); },
            { enableHighAccuracy: true, timeout: 10000, maximumAge: 5000 }
        );
    }

    async function sendLocationToServer(lat, lng) {
        if(!currentUser || currentUser.isAdmin) return;
        try {
            await fetch(API, {
                method: 'POST',
                body: JSON.stringify({
                    action: "updateLocation",
                    phone: currentUser.phone,
                    password: currentUser.password,
                    lat: lat,
                    lng: lng
                })
            });
        } catch(e) { /* silent */ }
    }

    // periodic sync for normal user: fetch member list, update map, and also push location
    let syncInterval = null;
    async function syncUserData() {
        if(!currentUser || currentUser.isAdmin) return;
        try {
            // Send location first (latest)
            if(lastSentLocation.lat !== 0 || lastSentLocation.lng !== 0) {
                await sendLocationToServer(lastSentLocation.lat, lastSentLocation.lng);
            }
            // then fetch all approved members (to show on map)
            const res = await fetch(API + "?request=members&phone=" + encodeURIComponent(currentUser.phone));
            const members = await res.json();
            updateMapWithMembers(members);
        } catch(e) { console.warn("Sync error", e); }
    }

    function updateMapWithMembers(members) {
        if(!mapInstance) initMap();
        // members format: each object { name, phone, lat, lng, photo, status }
        // remove markers for users not present or status changed
        const currentPhones = new Set();
        members.forEach(m => {
            if(m.status === "Approved" && m.lat && m.lng && !isNaN(parseFloat(m.lat)) && !isNaN(parseFloat(m.lng))) {
                currentPhones.add(m.phone);
                const lat = parseFloat(m.lat);
                const lng = parseFloat(m.lng);
                let marker = mapMarkers.get(m.phone);
                if(!marker) {
                    const iconHtml = L.divIcon({
                        html: `<div style="background: #00f2ff; width: 12px; height: 12px; border-radius: 50%; border:2px solid white; box-shadow:0 0 12px cyan;"></div>`,
                        iconSize: [12,12],
                        className: 'custom-marker'
                    });
                    marker = L.marker([lat, lng], { icon: iconHtml }).addTo(mapInstance);
                    marker.bindPopup(`<b>${m.name || "Member"}</b><br>📍 live`);
                    mapMarkers.set(m.phone, marker);
                } else {
                    marker.setLatLng([lat, lng]);
                }
            } else {
                // if not approved or invalid coords, remove marker if exists
                if(mapMarkers.has(m.phone)) {
                    mapInstance.removeLayer(mapMarkers.get(m.phone));
                    mapMarkers.delete(m.phone);
                }
            }
        });
        // remove markers for phones no longer in list
        for(let [phone, marker] of mapMarkers.entries()) {
            if(!currentPhones.has(phone)) {
                mapInstance.removeLayer(marker);
                mapMarkers.delete(phone);
            }
        }
        // center map if first time?
        if(members.length && mapInstance.getCenter().lat === 30.37) {
            const firstValid = members.find(m => m.lat && m.lng);
            if(firstValid) mapInstance.setView([parseFloat(firstValid.lat), parseFloat(firstValid.lng)], 10);
        }
    }
